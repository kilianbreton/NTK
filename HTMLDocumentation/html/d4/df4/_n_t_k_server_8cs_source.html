694  break;

696  this.name = node.getValue();
697  break;
698  case "ctype":
699  this.ctype = setCtype(node.getValue());
700  break;

703  break;

705  bool.TryParse(node.getValue(), out plugins);
706  break;
707  case "seckey":
708  this.secKey = node.getValue();
709  break;
710  case "service":
711  stype = node.getAttibuteV("name");
712  break;

717  }
718  break;
719  case "logs":
720  if (node.isChildExist("NTK")) { this.logs = Log_NTK.getInstance(node.getChildV("NTK")); }
721  if (node.isChildExist("Database")) { this.database.Logs = Log_Database.getInstance(node.getChildV("Database")); }

763  break;

767  break;

770  bool end = false;

773  {
774  if (cpt == extServices.Count) { end = true; }
775  else if (stype.Equals(extServices[cpt].Config.name))
776  {
777  service = extServices[cpt];
778  find = true;
779  end = true;
780  Console.WriteLine(" [OK]");
781  }

783  }

787  Console.ForegroundColor = ConsoleColor.DarkRed;
788  Console.WriteLine(" [BAD]");
